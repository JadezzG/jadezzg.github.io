# Führen Sie diese Befehle auf Ihrem Linux-Server aus, um die Groß-/Kleinschreibung der Bilddateien zu korrigieren:

# Sichern Sie zuerst die vorhandenen Dateien (falls vorhanden)
cp tools/img/Oridecon.png tools/img/Oridecon.png.bak 2>/dev/null
cp tools/img/Elunium.png tools/img/Elunium.png.bak 2>/dev/null
cp tools/img/Bradium.png tools/img/Bradium.png.bak 2>/dev/null

# Umbenennen Sie die Dateien mit der richtigen Groß-/Kleinschreibung
mv tools/img/oridecon.png tools/img/Oridecon.png
mv tools/img/elunium.png tools/img/Elunium.png
mv tools/img/bradium.png tools/img/Bradium.png

# Weitere Dateien, die Sie eventuell auch umbenennen müssen:
mv tools/img/enriched-oridecon.png tools/img/Enriched-Oridecon.png 2>/dev/null
mv tools/img/enriched-elunium.png tools/img/Enriched-Elunium.png 2>/dev/null
mv tools/img/enriched-bradium.png tools/img/Enriched-Bradium.png 2>/dev/null
mv tools/img/hd-oridecon.png tools/img/HD-Oridecon.png 2>/dev/null

mv tools/img/uhd-oridecon.png tools/img/UHD-Oridecon.png 2>/dev/null
